1) Nas funções helper, garante que existem estas funções:

function isRuler() {
  return request.auth != null &&
  exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
  get(/databases/$(database)/documents/users/$(request.auth.uid)).data.estatuto == 'ruler';
}

function votingIsOpen() {
  return exists(/databases/$(database)/documents/settings/endateworld) &&
  request.time < get(/databases/$(database)/documents/settings/endateworld).data.end;
}

2) Garante que o documento settings/endateworld pode ser lido pelo site público:

match /settings/endateworld {
  allow read: if true;
  allow write: if isAdmin();
}

3) Substitui os blocos worldcupextra e worldcupextraVisitors por estes:

match /worldcupextra/{docId} {
  allow read: if true;

  // SECÇÃO 1 — prognóstico inicial, antes da data limite.
  allow create: if
    (
      votingIsOpen() &&
      request.resource.data.status != 'official' &&
      request.resource.data.status != 'section2' &&
      request.resource.data.participantKey == docId &&
      request.resource.data.participantName is string &&
      request.resource.data.participantName.size() >= 2 &&
      request.resource.data.visitorKey is string &&
      !exists(/databases/$(database)/documents/worldcupextra/$(docId)) &&
      !exists(/databases/$(database)/documents/worldcupextraVisitors/$(request.resource.data.visitorKey)) &&
      existsAfter(/databases/$(database)/documents/worldcupextraVisitors/$(request.resource.data.visitorKey))
    )
    ||
    // RESULTADOS OFICIAIS MANUAIS — só ruler.
    (
      isRuler() &&
      request.resource.data.status == 'official' &&
      request.resource.data.type == 'officialResult' &&
      request.resource.data.matchId is int
    )
    ||
    // RESULTADOS OFICIAIS AUTOMÁTICOS DA API — opção simples para o index.html conseguir sincronizar.
    // Nota: para máxima segurança, o ideal é trocar isto por uma Cloud Function agendada.
    (
      request.resource.data.status == 'official' &&
      request.resource.data.type == 'officialResult' &&
      request.resource.data.source == 'api' &&
      request.resource.data.matchId is int &&
      request.resource.data.homeTeam is string &&
      request.resource.data.awayTeam is string &&
      request.resource.data.homeGoals is int &&
      request.resource.data.awayGoals is int &&
      request.resource.data.finished == true
    )
    ||
    // SECÇÃO 2 — reformulação das eliminatórias.
    (
      request.resource.data.status == 'section2' &&
      request.resource.data.type == 'knockoutRevision' &&
      request.resource.data.participantKey is string &&
      request.resource.data.participantName is string &&
      request.resource.data.participantName.size() >= 2 &&
      request.resource.data.matchId is int &&
      request.resource.data.stage in ['round32', 'round16', 'quarterfinals', 'semifinals', 'third_place', 'final']
    );

  allow update: if
    (
      isRuler() &&
      request.resource.data.status == 'official' &&
      request.resource.data.type == 'officialResult' &&
      request.resource.data.matchId is int
    )
    ||
    // Permite que a API atualize um resultado oficial automático já existente.
    // Nota: para máxima segurança, usar Cloud Function.
    (
      request.resource.data.status == 'official' &&
      request.resource.data.type == 'officialResult' &&
      request.resource.data.source == 'api' &&
      request.resource.data.matchId is int &&
      request.resource.data.homeTeam is string &&
      request.resource.data.awayTeam is string &&
      request.resource.data.homeGoals is int &&
      request.resource.data.awayGoals is int &&
      request.resource.data.finished == true
    )
    ||
    (
      request.resource.data.status == 'section2' &&
      request.resource.data.type == 'knockoutRevision' &&
      request.resource.data.participantKey is string &&
      request.resource.data.participantName is string &&
      request.resource.data.participantName.size() >= 2 &&
      request.resource.data.matchId is int &&
      request.resource.data.stage in ['round32', 'round16', 'quarterfinals', 'semifinals', 'third_place', 'final']
    );

  allow delete: if isRuler();
}

match /worldcupextraVisitors/{visitorKey} {
  allow read: if isAdmin();

  allow create: if
    votingIsOpen() &&
    request.resource.data.participantKey is string &&
    request.resource.data.participantName is string &&
    !exists(/databases/$(database)/documents/worldcupextraVisitors/$(visitorKey)) &&
    existsAfter(/databases/$(database)/documents/worldcupextra/$(request.resource.data.participantKey));

  allow update, delete: if isAdmin();

// --- SECÇÃO 2: Reformulações com janelas temporais ---
function reformWindowIsOpen(stage) {
  let w = get(/databases/$(database)/documents/settings/worldcupReformWindows).data;
  return exists(/databases/$(database)/documents/settings/worldcupReformWindows) && (
    (stage == 'round32' && request.time >= w.dezasseisAvosOpen && request.time < w.dezasseisAvosClose) ||
    (stage == 'round16' && request.time >= w.oitavosOpen && request.time < w.oitavosClose) ||
    (stage == 'quarterfinals' && request.time >= w.quartosOpen && request.time < w.quartosClose) ||
    (stage == 'semifinals' && request.time >= w.meiasOpen && request.time < w.meiasClose) ||
    (stage == 'third_place' && request.time >= w.terceiroLugarOpen && request.time < w.terceiroLugarClose) ||
    (stage == 'final' && request.time >= w.finalOpen && request.time < w.finalClose)
  );
}

match /worldcupextraReforms/{docId} {
  allow read: if true;
  allow create, update: if request.resource.data.status == 'section2'
    && request.resource.data.type == 'knockoutRevision'
    && request.resource.data.participantKey is string
    && request.resource.data.participantName is string
    && request.resource.data.stage is string
    && request.resource.data.matchId is int
    && reformWindowIsOpen(request.resource.data.stage);
  allow delete: if isAdmin();
}

// Horários oficiais dos jogos usados para bloquear escolhas de marcadores/battles.
// Colocar dentro de match /databases/{database}/documents { ... }
function matchKickoffPassed(matchId) {
  return exists(/databases/$(database)/documents/worldcupextraMatches/$(matchId < 10 ? 'match_00' + string(matchId) : matchId < 100 ? 'match_0' + string(matchId) : 'match_' + string(matchId)))
    && request.time >= get(/databases/$(database)/documents/worldcupextraMatches/$(matchId < 10 ? 'match_00' + string(matchId) : matchId < 100 ? 'match_0' + string(matchId) : 'match_' + string(matchId))).data.kickoff + duration.value(1, 's');
}

match /worldcupextraMatches/{matchDocId} {
  allow read: if true;
  allow create: if isAdmin() && (
    matchDocId.matches('^match_[0-9]{3}$') ||
    (matchDocId.matches('^official-match-[0-9]+$') && request.resource.data.matchId is int && matchKickoffPassed(request.resource.data.matchId))
  );
  allow update, delete: if isAdmin() && resource.data.matchId is int && matchKickoffPassed(resource.data.matchId);
}

// Escolhas de marcador nas Battles.
// Esta regra é uma base. A versão final pode apertar mais quando a estrutura das battles estiver fechada.
match /worldcupextraBattleScorers/{pickId} {
  allow read: if true;

  // O jogador pode criar a sua escolha antes do jogo começar, usando PIN convertido em pinHash.
  // O PIN real NÃO deve ser gravado neste documento.
  //
  // Documento esperado:
  // matchDocId: "match_001"
  // matchId: 1
  // battleId: "live_match_001_battle_01"
  // participantKey: "andre-henriques"
  // participantName: "André Henriques"
  // pinHash: "..."
  // pickedPlayerName: "Cristiano Ronaldo"
  // pickedPlayerTeam: "Portugal"
  // stage: "round32"
  allow create: if
    request.resource.data.matchDocId is string
    && exists(/databases/$(database)/documents/worldcupextraMatches/$(request.resource.data.matchDocId))
    && request.time < get(/databases/$(database)/documents/worldcupextraMatches/$(request.resource.data.matchDocId)).data.kickoff
    && request.resource.data.participantKey is string
    && exists(/databases/$(database)/documents/worldcupextra/$(request.resource.data.participantKey))
    && request.resource.data.pinHash is string
    && request.resource.data.pinHash == get(/databases/$(database)/documents/worldcupextra/$(request.resource.data.participantKey)).data.pinHash
    && request.resource.data.battleId is string
    && request.resource.data.pickedPlayerName is string
    && request.resource.data.pickedPlayerName.size() >= 2
    && request.resource.data.pickedPlayerTeam is string
    && !('pin' in request.resource.data)
    && !exists(/databases/$(database)/documents/worldcupextraBattleScorers/$(pickId));

  allow update, delete: if isAdmin();
}
